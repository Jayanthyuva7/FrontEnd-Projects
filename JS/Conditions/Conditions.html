Colors using Conditions

  • Exceeds Expectations
  • Meets Expectations
  • Needs Improvement

Code Used

<script>
document.addEventListener('DOMContentLoaded', function() {
const conditions = document.querySelectorAll('ul li');
conditions.forEach(condition => {
if (condition.textContent.includes('Exceeds Expectations')) {
condition.style.color = 'goldenrod';
} else if (condition.textContent.includes('Meets Expectations')) {
condition.style.color = 'green';
} else {
condition.style.color = 'red';
}
});
});
</script>
